//All player actors uses different weapons (mostly) so we had to do this

class Boolprop : Inventory
{
	Default
	{
		Inventory.MaxAmount 1;
		+Inventory.UNDROPPABLE;
		-Inventory.INVBAR;
	}
}

class IsSmileyHead : Boolprop {}
class IsUncleSmiley : Boolprop {}
class IsEmoley : Boolprop {}
class Buddy : Boolprop {}

class ChainsawPickup : CustomInventory replaces Chainsaw
{
	Default
	{
		Inventory.PickupMessage "$GOTCHAINSAW";
		Inventory.PickupSound "misc/w_pkup";
	}
  States
  {
  Spawn:
    CSAW A -1;
    loop;
	
  Pickup:
	TNT1 A 0 A_IfCheckWeapon;
	stop;
	
  }
}

extend class ChainsawPickup
{
	action void A_IfCheckWeapon()
	{
		if (CheckInventory("IsSmileyHead", 1))
		{
			A_GiveInventory("Toothbrush", 1);
		}
		else
			if (CheckInventory("IsUncleSmiley", 1))
			{
				A_GiveInventory("Toothbrush", 1);
			}
	}
}
